Perfect — we can do that with nvim -d (diff mode) or vimdiff and git show to extract the files.

gdiffedit() {
    local branch1 branch2 file1 file2 tmp1 tmp2 editor

    # Choose first branch
    echo "Select the first branch:"
    select branch1 in $(git branch --all | sed 's/^[* ]*//'); do
        [ -n "$branch1" ] && break
    done

    # Choose file from first branch
    echo "Select a file from $branch1:"
    select file1 in $(git ls-tree -r --name-only "$branch1"); do
        [ -n "$file1" ] && break
    done

    # Choose second branch
    echo "Select the second branch:"
    select branch2 in $(git branch --all | sed 's/^[* ]*//'); do
        [ -n "$branch2" ] && break
    done

    # Choose file from second branch
    echo "Select a file from $branch2:"
    select file2 in $(git ls-tree -r --name-only "$branch2"); do
        [ -n "$file2" ] && break
    done

    # Create temporary files
    tmp1=$(mktemp /tmp/gitfile1.XXXX)
    tmp2=$(mktemp /tmp/gitfile2.XXXX)

    # Extract files into temp files
    git show "$branch1:$file1" > "$tmp1"
    git show "$branch2:$file2" > "$tmp2"

    # Open in Neovim diff mode (editable)
    editor="${EDITOR:-nvim}"  # default to nvim if $EDITOR not set
    "$editor" -d "$tmp1" "$tmp2"

    # Ask if you want to copy changes back to working directory
    read -p "Do you want to save changes back to working tree? [y/N] " save_changes
    if [[ "$save_changes" =~ ^[Yy]$ ]]; then
        cp "$tmp1" "$file1"
        cp "$tmp2" "$file2"
        echo "Changes saved to $file1 and $file2"
    fi

    # Cleanup
    rm -f "$tmp1" "$tmp2"
}

myMethod() {
  local output_dir=""
  local selections=()

  # -------------
  # Validate output directory
  # -------------
  if [ -n "$1" ]; then
    if [ ! -d "$1" ]; then
      echo "Invalid directory: $1"
      return 1
    fi
    output_dir="$(cd "$1" && pwd)"
  fi

  # -------------
  # Branch selection loop
  # -------------
  while true; do
    local branch
    branch=$(git branch --all --format='%(refname:short)' \
      | sed 's|^origin/||' \
      | sort -u \
      | fzf --prompt="Select branch (ESC to finish): ")

    [ -z "$branch" ] && break

    # -------------
    # File selection for branch
    # -------------
    local files
    files=$(git ls-tree -r --name-only "$branch" \
      | fzf --multi --prompt="Select files from $branch (ESC to skip): ")

    [ -z "$files" ] && continue

    while IFS= read -r file; do
      selections+=("$branch:$file")
    done <<< "$files"
  done

  # -------------
  # No selections → nothing to do
  # -------------
  if [ "${#selections[@]}" -eq 0 ]; then
    echo "No files selected. Nothing to do."
    return 0
  fi

  # -------------
  # Create files
  # -------------
  for entry in "${selections[@]}"; do
    local branch="${entry%%:*}"
    local file="${entry#*:}"

    local dir base name ext target_dir target_file

    dir="$(dirname "$file")"
    base="$(basename "$file")"
    name="${base%.*}"
    ext="${base##*.}"

    if [ "$name" = "$ext" ]; then
      ext=""
    else
      ext=".$ext"
    fi

    if [ -n "$output_dir" ]; then
      target_dir="$output_dir"
    else
      target_dir="$dir"
      mkdir -p "$target_dir"
    fi

    target_file="${target_dir}/${name}_${branch}_temp${ext}"

    if [ -e "$target_file" ]; then
      echo "Skipping existing file: $target_file"
      continue
    fi

    git show "$branch:$file" > "$target_file" || {
      echo "Failed to extract $file from $branch"
      continue
    }

    echo "Created: $target_file"
  done
}
